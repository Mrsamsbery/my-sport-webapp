// Обновленная функция отправки напрямую на бэкенд без сторонних фотохостингов async function startAnalysis() { if (!selectedFile) return; const loader = document.getElementById("loader"); const errorBlock = document.getElementById("error-block"); const resultsPanel = document.getElementById("results-panel"); const analyzeBtn = document.getElementById("analyze-btn"); loader.style.display = "block"; errorBlock.style.display = "none"; resultsPanel.style.display = "none"; analyzeBtn.disabled = true; try { document.getElementById("loader-text").innerText = "Передача файла на сервер аналитики ИИ..."; // Создаем FormData и кладем туда файл и параметры текстом const formData = new FormData(); formData.append("image", selectedFile); formData.append("user_id", String(userId)); formData.append("mode", currentMode); // Отправляем данные напрямую во Flask const response = await fetch(`${BACKEND_URL}/api/process`, { method: "POST", headers: { "ngrok-skip-browser-warning": "true" // Для работы через Ngrok туннель }, body: formData // Передаем formData (браузер сам выставит нужные boundary) }); if (!response.ok) { const errorData = await response.json(); throw new Error(errorData.error || `Сбой бэкенда: ${response.status}`); } const data = await response.json(); if (data.success && data.analysis) { // Раскладываем структурированные данные из JSON OpenAI по фиолетовым карточкам UI document.getElementById("ui-sport").innerText = data.analysis.sport || "⚽"; document.getElementById("ui-match-title").innerText = data.analysis.match_title || "Матч не определен"; document.getElementById("ui-league").innerText = data.analysis.league || "Турнир не определен"; document.getElementById("ui-capper-pick").innerText = data.analysis.capper_pick || "-"; document.getElementById("ui-value-verdict").innerText = data.analysis.value_verdict || "-"; document.getElementById("ui-value-profit").innerText = data.analysis.value_profit || "0/10"; document.getElementById("ui-margin-eval").innerText = data.analysis.margin_eval || "0%"; document.getElementById("ui-form-trends").innerText = data.analysis.form_trends || "-"; document.getElementById("ui-head-to-head").innerText = data.analysis.head_to_head || "-"; document.getElementById("ui-final-verdict").innerText = data.analysis.final_verdict || "-"; // Запускаем анимацию круга процентов updateConfidenceCircle(data.analysis.confidence_percent); // Показываем блок результатов, обновляем баланс лимитов resultsPanel.style.display = "block"; fetchUserLimits(); } else { throw new Error("Неверный формат ответа от искусственного интеллекта."); } } catch (err) { console.error(err); errorBlock.innerText = `Ошибка: ${err.message || "Ошибка связи с сервером аналитики."}`; errorBlock.style.display = "block"; } finally { loader.style.display = "none"; analyzeBtn.disabled = false; } }
